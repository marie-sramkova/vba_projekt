CREATE TABLE BOOK (
                      isbn int primary key,
                      name varchar(50) not null,
                      number_of_pages int,
                      genre varchar(50),
                      content varchar(1000),
                      image_url varchar(20000),
                      binding varchar(50));

create table author (
                        author_id int primary key generated always as identity,
                        first_name varchar(50),
                        surname varchar(50) not null,
                        birth_day date);

create table appUser(
                        app_user_id int primary key generated always as identity,
                        nick_name varchar(50) not null,
                        password_hash blob not null,
                        salt blob not null
);

create table Enrollment(
                           enrollment_id int primary key generated always as identity,
                           app_user_id int not null references appUser(app_user_id),
                           book_isbn int not null references BOOK(isbn)
);

create table Ownership (ownership_id int primary key generated always as identity,
                        book_isbn int not null references BOOK(isbn),
                        author_id int not null references author(author_id)
);


ALTER TABLE Ownership ADD CONSTRAINT UQ_bookIsbn_authorId UNIQUE(book_isbn, author_id);
